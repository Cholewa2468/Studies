    public class Osoba
    {

        protected string imie;
        //private string nazwisko;
        protected string nazwisko;
        
        public string Imie
        {
            get => this.imie;
            set => this.imie = value;
        }
        
        public string Nazwisko
        {
            get => this.nazwisko;
            set => this.nazwisko = value;
        }
        
        public void UstawImie(string imie) //Set
        {
            this.imie = imie;    
        }
        
        public string PobierzImie() //Get
        {
            return this.imie;
        }
        
        public virtual void UstawNazwisko(string nazwisko) //Set
        {
            this.nazwisko = nazwisko;    
        }
        
        public virtual string PobierzNazwisko() //Get
        {
            return this.nazwisko;
        }
        
        public Osoba()
        {
            this.imie = "";
            this.nazwisko = "";
        }
        
        public Osoba(string imie, string nazwisko)
        {
            UstawImie(imie);
            UstawNazwisko(nazwisko);
        }
    }
